B - Average

题目大意：

n个评委评分，问存在多少种方案使得评委评分的平均分等于其中一个评委的评分。（每个评委都是独立的）

输入：n个评委，full 每个评委给的最高分。

思路：

有容斥和dp前缀和优化两种做法，以下是基于容斥的思考

首先很容易可以知道：

    在无限制条件的方案数量

根据条件  

枚举每个可能存在的a，用容斥原理（原理参考组合数学P107）求出每个平均值对应的方案。（有一个注意的点是每个评委都是独立的，所以需要每次都乘以n）

#include <bits/stdc++.h>
 
using namespace std;
 
const long long MOD = 1e9 + 7;
 
const int MAX_N = 60;
const int MAX_F = 200;
const int MAX_NF = MAX_N * MAX_F;
 
//      [fullscore][score]     [teachers]
int sums[MAX_F + 1][MAX_NF + 1][MAX_N + 1];
 
void precalc() {
    for (int full = 1; full <= MAX_F; full++) {
        for (int i = 0; i <= full; i++) {
            sums[full][i][1] = 1;
        }
 
        for (int j = 2; j <= MAX_N; j++) {
            long long s = 0;
            for (int i = 0; i <= full * j; i++) {
                s += sums[full][i][j - 1];
                s %= MOD;
 
                if (i - full - 1 >= 0) {
                    s -= sums[full][i - full - 1][j - 1];
                    if (s < 0) s += MOD;
                }
 
                sums[full][i][j] = int(s);
            }
        }
    }
}
 
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
 
    precalc();
 
    int n, f;
    cin >> n >> f;
 
    while (n || f) {
        long long ans = 0;
        for (long long x = 0; x <= f; x++) {
            ans += (long long) n * sums[f][(n - 1) * x][n - 1];
            ans %= MOD;
        }
 
        cout << ans << endl;
 
        cin >> n >> f;
    }
}


F-Dictionary Game

题目大意：

给一棵可以变化的trie，在trie上选一个点，删除的是它以及它的孩子节点，两人轮流操作，不能操作的为输。

思路：

树上删边游戏。

树上删边游戏的模型是这样的：📎树上删边游戏.doc          参考博客

树上删边游戏博弈结论：叶子节点的SG值为0，非叶子节点的SG值为子节点SG值+1的异或和。

每次加入一个字符串便在Trie树上扩展一条链，更新链上节点的SG值（注意消除这条链上之前节点的影响）

upd：这道题是树上公平博弈 ，也有不公平的   有机会研究一下。


#include <bits/stdc++.h>
using namespace std;
const int maxn=4000005;
struct Trie{
    int ch[maxn][26];
    int sg[maxn];
    int sz;
    int newnode(){
        memset(ch[sz],-1,sizeof(ch[sz]));
        sg[sz]=1;
        return sz++;
    }
    void init(){
        sz=0;
        newnode();
    }
    void insert(char s[]){
        int len=strlen(s);
        int u=0;
        int path[41];
        for(int i=0;i<len;i++){
            path[i]=u;
            int c=s[i]-'a';
            sg[u]--;
            if(ch[u][c]==-1)ch[u][c]=newnode();
            else sg[u]^=sg[ch[u][c]];
            u=ch[u][c];
        }
        for(int i=len-1;i>=0;i--){
            u=path[i];
            int v=ch[u][s[i]-'a'];
            sg[u]^=sg[v];
            sg[u]++;
        }
    }
}trie;
char str[50];
int main(){
    int T;
    scanf("%d",&T);
    for(int tt=1;tt<=T;tt++){
        int n;
        scanf("%d",&n);
        trie.init();
        for(int i=1;i<=n;i++){
            scanf("%s",str);
            trie.insert(str);
        }
        printf("Case %d:\n",tt);
        int q;
        scanf("%d",&q);
        for(int i=1;i<=q;i++){
            scanf("%s",str);
            trie.insert(str);
            int p=trie.sg[0]-1;
            if(p==0)printf("2\n");
            else printf("1\n");
        }
    }
}


H - Witcher Potion

题目大意

勇者打怪

  • 勇士的属性 : 血条初始为 100 , 不能 <0 . 毒条初始为 0 , 不能 >100
  • 每个药可以 : 增加  点血条，增加 点毒条
  • 每打一个怪 : 降低 k 点血条, 降低 m 点毒条

求最大的打怪数量

数据范围 :

思路

暴力压状dp

  • 复杂度 : 


#include <bits/stdc++.h>
using namespace std;
//    药的状态 hp   毒条
int dp[1 << 9][101][101];
int e[9], p[9], k, m, n;
int main(){
    int T; scanf("%d", &T);
    while(T--){
        scanf("%d%d", &k, &m);
        scanf("%d", &n);
        for(int i = 1; i <= n; i++) scanf("%d", e + i);
        for(int i = 1; i <= n; i++) scanf("%d", p + i);
        int len = (1 << n);
        for(int i = 0; i < len; i++)
            for(int j = 0; j <= 100; j++)
                for(int k = 0; k <= 100; k++)
                    dp[i][j][k] = -1e9;
        dp[0][100][0] = 0;
        for(int i = 0; i < len; i++){
            for(int j = 100; j >= k + 1; j--){
                for(int u = 0; u <= 99; u++){
                    //不嗑药
                    int& bky = dp[i][j - k][max(0, u - m)];
                    bky = max(dp[i][j][u] + 1, dp[i][j - k][max(0, u - m)]);
                    //嗑药
                    for (int f = 1; f <= n; f++) {
                        if (((1 << (f - 1)) & i) == 0 && max(0, u - m) + p[f] <= 99) {
                            int& ky = dp[i + (1 << (f - 1))][min(100, j - k + e[f])][max(0, u - m) + p[f]];
                            ky = max(ky, dp[i][j][u] + 1);
                        }
                    }
                }
            }
        }
        int ans = 0;
        for(int i = 0; i < len; i++)
            for(int j = 0; j <= 100; j++)
                for(int u = 0; u <= 99; u++)
                    ans = max(ans, dp[i][j][u]);
        printf("%d\n", ans);
    }
    return 0;
}



I - Sky Tax

题目大意

求一个可换根树的某一个节点的子树size

  • 操作1 : 换根
  • 操作2 : 求子树size

思路

  1. 首先以1为根建树, 记录每个节点的子树数量为 size[] 
  2. 记录操作1换的"根"为 rt 
  3. 查询x子树大小时
    1. 如果 rt 在 x 子树外则答案为size[x]
    2. 如果 rt 在 x 子树内则答案为 n-size[d] , 其中d为从 rt 到 x 的路径中距离 x 最近的那个节点
  • 利用树链剖分判读子树和lca2


#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 50;
struct edge {
    int to, nx;
} e[N * 2];
int n, q, r, ecnt, idc;
int head[N], dep[N], siz[N], son[N], top[N], f[N], st[N], en[N], id[N];
void add(int x, int y) {
    e[++ecnt].to = y;
    e[ecnt].nx = head[x];
    head[x] = ecnt;
}
void dfs1(int x) {
    siz[x] = 1;
    dep[x] = dep[f[x]] + 1;
    for (int i = head[x]; i; i = e[i].nx) {
        int nd = e[i].to;
        if (nd == f[x]) continue;
        f[nd] = x;
        dfs1(nd);
        siz[x] += siz[nd];
        if (!son[x] || siz[son[x]] < siz[nd]) son[x] = nd;
    }
}
void dfs2(int x, int tp) {
    id[x] = ++idc;
    st[x] = idc;
    top[x] = tp;
    if (son[x]) dfs2(son[x], tp);
    for (int i = head[x]; i; i = e[i].nx) {
        int nd = e[i].to;
        if (nd == f[x] || nd == son[x]) continue;
        dfs2(nd, nd);
    }
    en[x] = idc;
}
//使用条件 y 在 x的子树内
int lca2(int x, int y) {
    int t;
    while (top[x] != top[y]) t = top[y], y = f[top[y]];
    return x == y ? t : son[x];
}

int main() {
    int T;
    scanf("%d", &T);
    for (int C = 1; C <= T; C++) {
        printf("Case #%d:\n", C);
        ecnt = idc = 0;
        for (int i = 0; i <= n; i++) {
            head[i] = son[i] = f[i] = siz[i] = 0;
        }
        scanf("%d%d%d", &n, &q, &r);
        int a, b;
        for (int i = 0; i < n - 1; i++) {
            scanf("%d%d", &a, &b);
            add(a, b);
            add(b, a);
        }
        dfs1(1);
        dfs2(1, 1);
        int ope, x;
        while(q--){
            scanf("%d%d", &ope, &x);
            if(ope == 0) r = x;
            if(ope == 1){
                if(x == r){
                    printf("%d\n", n);
                }else if(st[x] <= id[r] && id[r] <= en[x]){
                    //说明r在x的子树内
                    printf("%d\n", n - siz[lca2(x, r)]);
                }else{
                    printf("%d\n", siz[x]);
                }
            }
        }
    }
    return 0;
}


D. Find C

题目大意

  • 给定一条线段的A，B端点，求得k个点且点的要求如下：
  1. 能与AB形成三角形
  2. 除了AB边上可以有整点（横坐标和纵坐标同为整数的点），其余边及形成的三角形内部均不包含整点

思路

首先可以泛泛想到对于这条边来说若在其上下画上一条平行边，显然可以得到在两边分别于原直线形成的带状区域中是可能含有整点，并且整点的分布是有规律的，所以我们可以联想到找到距离原直线最近的点，因此再过该点做平行线，由于该点最近，所以在带状区域中是不含有整点的，因此在CB及CA的边上也是同样不含整点的。因此，题目至此已经化简到寻找到距离已知直线最近的点的位置，随后只要沿着过该点的平行线直接朝一个方向扩展k个点即可（由于本题给了1e14并且原点AB的范围在1e9内，所以并不需要像Claris分两头找，直接暴力一个方向即可）。

对于求解特殊点，利用exgcd找到与已知直线斜率最接近的点，极小化求解逆元得到相应的x，y，最后确定好点的位置，扩展时x的步长即为，y的步长为。


#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int T,k;
ll xa,ya,xb,yb;
ll A,B,x,y;
 
inline void exgcd(ll a, ll b, ll& x, ll& y) {if (b == 0) {x = 1, y = 0;return;}exgcd(b, a % b, y, x);y -= a / b * x;}
 
int main() {
    //freopen("in.txt", "r", stdin);
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin>>T;
    while (T--) {
        cin >> xa >> ya >> xb >> yb >> k;
        A = ya - yb;
        B = xb - xa;
        exgcd(abs(A), abs(B), x, y);
        if (A < 0) x = -x;
        if (B < 0) y = -y;
        x += xb, y += yb;
        ll G = __gcd(abs(A), abs(B));
        ll dx = (xa - xb) / G;
        ll dy = (ya - yb) / G;
        while (k--) {
            cout << x << " " << y << endl;
            x += dx;
            y += dy;
        }
    }
}


E. ACM Tax

题目大意

  • 给定一棵树，并且给出每条边的权重
  • 给出Q（1<=Q<=1e6）次查询，每次给出u，v
  • 求从u到v的路径所经过的边经过排序后的中位数

思路

  1.  由于问题只需要求得中位数的值，因此可以联想到主席树板子功能（对于一个序列求得任意区间的第k大）
  2.  对于k值，我们利用述链剖分的思想直接求得从u到v多经过点的个数

得到序列的长度后，对于奇数的长度只需求第个值，对于偶数分别求得第个和第个的值除2.

  1. 而这个题最关键的地方我感觉是这个在插入值的方法很巧妙

     

     直接将对应节点的儿子节点加在自己父节点之下，后在查询时直接根据之前的性质不断二分下去找       到对应的节点即为所要找的点。

  1. 另外Claris树剖板子可以学一手


#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int N = 5e4+10, M = N*20;
int n;
int g[N],v[N<<1],w[N<<1],nxt[N<<1],ed;
int f[N],d[N],sz[N],son[N],top[N];
int tot,rt[N],l[M],r[M],val[M];

inline void add(int x, int y, int z){
    v[++ed]=y;
    w[ed]=z;
    nxt[ed]=g[x];
    g[x]=ed;
}
inline int ins(int x, int a, int b, int c){
    int y = ++tot;
    val[y]=val[x]+1;//该节点儿子儿子个数++
    if(a==b) return y;
    int mid=(a+b)>>1;
    if(c<=mid){
        l[y]=ins(l[x],a,mid,c);
        r[y]=r[x];
    }else{
        r[y]=ins(r[x],mid+1,b,c);
        l[y]=l[x];
    }
    return y;
}
void dfs(int x){
    sz[x]=1;
    for(int i = g[x];i;i=nxt[i])if(v[i]!=f[x]){
        f[v[i]]=x,d[v[i]]=d[x]+1;
        rt[v[i]]=ins(rt[x],1,100000,w[i]);
        dfs(v[i]),sz[x]+=sz[v[i]];
        if(sz[v[i]]>sz[son[x]])son[x]=v[i];
    }
}
void dfs2(int x, int y){
    top[x]=y;
    if(son[x])dfs2(son[x],y);
    for(int i = g[x];i;i=nxt[i])if(v[i]!=son[x]&&v[i]!=f[x])dfs2(v[i],v[i]);
}
inline int lca(int x, int y){
    for(;top[x]!=top[y];x=f[top[x]])if(d[top[x]]<d[top[y]])swap(x,y);
    return d[x]<d[y]?x:y;
}
inline int kth(int x, int y, int z, int k){
    int a=1,b=100000,mid,t;
    while(a<b){
        mid=(a+b)>>1;
        t=val[l[x]]+val[l[y]]-2*val[l[z]];
        if(k<=t){
            b=mid;
            x=l[x];
            y=l[y];
            z=l[z];
        }else{
            k-=t;
            a=mid+1;
            x=r[x];
            y=r[y];
            z=r[z];
        }
    }
    return a;
}
inline int query(int x, int y){
    int z = lca(x,y);
    int all=d[x]+d[y]-d[z]*2;
    if(all&1)return kth(rt[x],rt[y],rt[z],(all+1)/2)*2;
    return kth(rt[x],rt[y],rt[z],all/2)+kth(rt[x],rt[y],rt[z],all/2+1);
}

int main() {
    int _;
    scanf("%d", &_);
    while (_--) {
        cin >> n;
        ed = 0;
        for (int i = 1; i <= n; i++)g[i] = f[i] = d[i] = sz[i] = son[i] = top[i] = rt[i] = 0;
        for (int i = 1; i <= n-1; i++) {
            int x, y, z; scanf("%d%d%d",&x,&y,&z);
            add(x, y, z), add(y, x, z);
        }
        dfs(1);
        dfs2(1, 1);
        int m;scanf("%d",&m);
        for (int i = 1; i <= m; i++) {
            int x, y; scanf("%d%d",&x,&y);
            printf("%.1f\n", 0.5 * query(x, y));
        }
        for (int i = 1; i <= tot; i++) l[i] = r[i] = val[i] = 0;
        tot = 0;
    }
    return 0;
}
